\begin{figure}[h!]
    \subsection{Sestavení rovnic}
    \begin{gather*}
        U_A (G_1 + G_2 + G_3) + U_B (-G_2) + 0 = I \\
        U_A (-G_2) + U_B (G_2 + G_4) + U_C (-G_4) = -I_2 \\
        0 + U_B (-G_4) + U_C (G_4 + G_5) = I_2 - I_1 \\
    \end{gather*}
\end{figure}

\begin{figure}[h!]
    \subsection{Sestavení do matice}
    \begin{gather*}
        \begin{pmatrix}
            G_1 + G_2 + G_3 & -G_2 & 0 \\
            -G_2 &  G_2 + G_4 & -G_4  \\
            0 & -G_4 & G_4 + G_5 \\
        \end{pmatrix}
        \times
        \begin{pmatrix}
            U_A \\
            U_B \\
            U_C
        \end{pmatrix}
        =
        \begin{pmatrix}
            I \\
            -I_2 \\
            I_2 - I_1
        \end{pmatrix}
    \end{gather*}
\end{figure}
